In [9]:
# plotting a histogram
fig, ax = plt.subplots(1, 1)
ax.hist(d, 50, density=True)   # d is the distribution defined above, 50 is the # of bins in the histogram, and density=True returns the probability density
plt.tick_params(labelsize = 24)
plt.xlim([4.95,5.05])
x = np.linspace(4.95, 5.05, 1000)
ax.plot(x, stats.norm.pdf(x, loc = 5., scale = 0.01), linewidth = 8, alpha = 0.7)  # alpha designates opaqueness of curve
plt.show()
